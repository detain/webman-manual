# هيكل الدليل
```
.
├── app                           دليل التطبيق
│   ├── controller                دليل التحكم
│   ├── model                     دليل النموذج
│   ├── view                      دليل العرض
│   ├── middleware                دليل الوسيطة
│   │   └── StaticFile.php        وسيطة ملفات ثابتة المدمجة
|   └── functions.php             يتم كتابة الدوال المخصصة للأعمال في هذا الملف
|
├── config                        دليل الإعدادات
│   ├── app.php                   إعدادات التطبيق
│   ├── autoload.php              يتم تحميل الملفات المُعدة هنا تلقائياً
│   ├── bootstrap.php             تكوين الوظيفة التي تُشغل عند بدء التشغيل للعملية onWorkerStart
│   ├── container.php             تكوين الحاوية
│   ├── dependence.php            تكوين الاعتمادات للحاوية
│   ├── database.php              إعدادات قاعدة البيانات
│   ├── exception.php             إعدادات الاستثناءات
│   ├── log.php                   إعدادات السجل
│   ├── middleware.php            إعدادات الوسيطة
│   ├── process.php               تكوين العمليات المخصصة
│   ├── redis.php                 إعدادات ريديس
│   ├── route.php                 إعدادات الطريق
│   ├── server.php                إعدادات الخادم مثل المنافذ وعدد العمليات
│   ├── view.php                  إعدادات العرض
│   ├── static.php                تكوين ملفات الثابت وإعدادات وسيطة ملفات ثابتة
│   ├── translation.php           إعدادات اللغات المتعددة
│   └── session.php               إعدادات الجلسة
├── public                        دليل الموارد الثابتة
├── process                       دليل العمليات المخصصة
├── runtime                       دليل تشغيل التطبيق، يحتاج إلى إذن الكتابة
├── start.php                     ملف تشغيل الخدمة
├── vendor                        دليل المكتبات الخارجية المثبتة بواسطة Composer
└── support                       التكيفات للمكتبات (بما في ذلك مكتبات الطرف الثالث)
    ├── Request.php               فئة الطلبات
    ├── Response.php              فئة الاستجابة
    ├── Plugin.php                نصب وإزالة الإضافات
    ├── helpers.php               دوال المساعدة (يرجى كتابة الدوال المخصصة للأعمال في app/functions.php)
    └── bootstrap.php             سكريبت التهيئة بعد بدء التشغيل
```
